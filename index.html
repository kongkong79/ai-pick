async function renderAnalysis() { const container = document.getElementById('analysis-list'); const loader = document.getElementById('loading-indicator'); const isVip = localStorage.getItem('isVipUser') === 'true'; const t = translations[currentLang]; if (loader) loader.style.display = 'block'; container.innerHTML = ''; try { // 캐시 방지 및 파일 로드 const response = await fetch('sports_data.xlsx?v=' + Date.now()); if (!response.ok) throw new Error('File not found'); const arrayBuffer = await response.arrayBuffer(); const data = new Uint8Array(arrayBuffer); const workbook = XLSX.read(data, {type: 'array'}); const sheet = workbook.Sheets[workbook.SheetNames[0]]; // header: 1로 가져오면 순수 배열로 가져옵니다. const rows = XLSX.utils.sheet_to_json(sheet, {header: 1}); // 필터링 최소화: 홈팀(1번)과 어웨이팀(2번) 이름만 있으면 무조건 통과 const filteredRows = rows.slice(1).filter(row => row[1] && row[2]); if (filteredRows.length === 0) { container.innerHTML = '

데이터를 찾을 수 없습니다. (Excel Row Empty)

'; return; } filteredRows.forEach(row => { // 엑셀 인덱스 강제 매핑 const time = row[0] || ''; const home = row[1] || ''; const away = row[2] || ''; const odds = row[3] || '0.00'; const pick = row[4] || '-'; const hitRate = parseFloat(row[5]) || 0; const winP = parseFloat(row[6]) || 0; const drawP = parseFloat(row[7]) || 0; const lossP = parseFloat(row[8]) || 0; const sample = row[11] || '0'; const isHighAcc = hitRate >= 70; const card = document.createElement('div'); // 적중률 80% 이상 VIP 잠금 처리 if (hitRate >= 80 && !isVip) { card.className = 'analysis-list-item vip-locked-card'; card.innerHTML = ` ⭐ VIP ONLY
🔒

Hit Rate: ${hitRate}%

${t.unlock}
🕒 ${time}

${home} vs ${away}

`; } else { card.className = `analysis-list-item ${isHighAcc ? 'high-accuracy-card' : ''}`; card.innerHTML = ` ${isHighAcc ? `${t.highAcc}` : ''}
🕒 ${time} ${t.sample}: ${sample}

${home} vs ${away}

${t.pick}: ${pick} (${odds})
Win ${winP}%
Draw ${drawP}%
Hit Rate: ${hitRate}% AI365PICK.com
`; } container.appendChild(card); }); } catch (e) { console.error(e); container.innerHTML = '

오류가 발생했습니다. 잠시 후 다시 시도해 주세요.

'; } finally { if (loader) loader.style.display = 'none'; } }
