async function renderAnalysis() { const container = document.getElementById('analysis-list'); const loader = document.getElementById('loading-indicator'); const isVip = localStorage.getItem('isVipUser') === 'true'; const t = translations[currentLang]; loader.style.display = 'block'; container.innerHTML = ''; try { // 1. 엑셀 파일 로드 (캐시 방지 v=Date.now 추가) const response = await fetch('sports_data.xlsx?v=' + Date.now()); const arrayBuffer = await response.arrayBuffer(); const data = new Uint8Array(arrayBuffer); const workbook = XLSX.read(data, {type: 'array'}); const sheet = workbook.Sheets[workbook.SheetNames[0]]; // 2. 데이터를 배열 형태로 가져오기 const rows = XLSX.utils.sheet_to_json(sheet, {header: 1}); // 3. 필터 조건 수정: 샘플 수가 1개 이상이면 무조건 표시 (기존 10개에서 완화) const filteredRows = rows.slice(1).filter(row => { const sampleCount = parseInt(row[11]); return row[1] && row[2] && sampleCount >= 1; }); if (filteredRows.length === 0) { container.innerHTML = '

분석 데이터가 없거나 조건에 맞는 경기가 없습니다.

'; return; } console.log("표시할 경기 수:", filteredRows.length); filteredRows.forEach(row => { // [데이터 매핑] 엑셀 열 순서 기준 // 0:Time, 1:Home, 2:Away, 3:Odds, 4:Recommendation, 5:HitRate, 6:Win%, 7:Draw%, 8:Loss%, 11:Sample const time = row[0] || ''; const home = row[1] || ''; const away = row[2] || ''; const odds = row[3] || '0.00'; const pick = row[4] || '-'; const hitRate = parseFloat(row[5]) || 0; const winP = parseFloat(row[6]) || 0; const drawP = parseFloat(row[7]) || 0; const lossP = parseFloat(row[8]) || 0; const sample = row[11] || '0'; const isHighAcc = hitRate >= 70; const card = document.createElement('div'); // VIP 로직: 적중률 80% 이상이면서 VIP가 아닌 경우 잠금 if (hitRate >= 80 && !isVip) { card.className = 'analysis-list-item vip-locked-card'; card.innerHTML = ` ⭐ VIP ONLY
🔒

Hit Rate: ${hitRate}%

${t.unlock}
🕒 ${time}

${home} vs ${away}

`; } else { card.className = `analysis-list-item ${isHighAcc ? 'high-accuracy-card' : ''}`; card.innerHTML = ` ${isHighAcc ? `${t.highAcc}` : ''}
🕒 ${time} ${t.sample}: ${sample}

${home} vs ${away}

${t.pick}: ${pick} (${odds})
Win ${winP}%
Draw ${drawP}%
Loss ${lossP}%
Hit Rate: ${hitRate}% AI365PICK.com
`; } container.appendChild(card); }); } catch (e) { console.error("렌더링 에러:", e); container.innerHTML = `

에러: ${e.message}

`; } finally { loader.style.display = 'none'; } }
